 *
 * @author Aeranythe Echosong
 */
public enum Tile implements java.io.Serializable{

    FLOOR((char) 250, AsciiPanel.green),

    WALL((char) 177, AsciiPanel.brightBlack),

    BOUNDS('x', AsciiPanel.magenta),

    GOAL((char) 3, AsciiPanel.brightRed),

    BOMB((char)42,AsciiPanel.red),

    Brick((char) 177, AsciiPanel.brightWhite);

    private char glyph;

    public char glyph() {
        return glyph;
    }

    private Color color;

    public Color color() {
        return color;
    }

    public void setColor (Color color){
        this.color = color;
    }

    public boolean isDiggable() {
        return this == Tile.Brick;
    }

    public boolean isGround() {
        return this == Tile.FLOOR;
    }

    Tile(char glyph, Color color) {
        this.glyph = glyph;
        this.color = color;
    }
}
